const mongoose = require("mongoose");
const Schema = mongoose.Schema;

const movieSchema = new Schema({
    movie_id: {
        type: Number,
        required: true,
    },
    title: {
        type: String,
        required: true,
    },
    cast: {
        type: String
    },
    crew: {
        type: String
    }
});

const Movie = mongoose.model("Movie", movieSchema);
module.exports = Movie;
